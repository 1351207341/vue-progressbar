Requirements

  • Vue.js ^1.0.0

Instllation

npm

$ npm install vue-progressbar

Usage

<!-- your component -->
<script>
import progress from 'vue-progressbar'

export default {
  data() {
      return {
        precent: 0,
        options: {
          color: '#73ccec',
          height: '2px'
        }
      }
    },
    components: {
      progress
    },
    ready() {
      setInterval(() => {
        this.precent++
          if (this.precent > 100) {
            this.precent = 0
          }
      }, 100)
    }
}
</script>

<template>
  <progress :precent="precent" :options="options" />
</template>
